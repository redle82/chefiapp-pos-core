import { getCoreSessionAsync } from '@/infra/auth/getCoreSessionAsync';

async function authHeaders(): Promise<Record<string, string>> {
  const session = await getCoreSessionAsync();
  const accessToken = session?.access_token;

  const baseHeaders: Record<string, string> = {
    apikey: CORE_ANON_KEY,
    'Content-Type': 'application/json',
    Accept: 'application/json',
  };

  if (accessToken) {
    return {
      ...baseHeaders,
      Authorization: `Bearer ${accessToken}`,
    };
  }

  // Fallback for public/unauthenticated flows
  return {
    ...baseHeaders,
    Authorization: `Bearer ${CORE_ANON_KEY}`,
  };
}

async run(...) {
  // ...
  const response = await fetch(url, {
    method,
    headers: await authHeaders(),
    body,
  });
  // ...
}

async rpc(...) {
  // ...
  const response = await fetch(url, {
    method: 'POST',
    headers: await authHeaders(),
    body,
  });
  // ...
}

// Any internal fetch wrapper function that used headers() now uses authHeaders and is async, for example:

async function internalFetchWrapper(...) {
  const response = await fetch(url, {
    headers: await authHeaders(),
    // other options
  });
  // ...
}
